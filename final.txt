    <script>
        function showSection(sectionId) {
            var sections = document.querySelectorAll('.section');
            var navLinks = document.querySelectorAll('nav ul li a');
            var temperatureContainer = document.querySelector('.temperature-container'); // Target temperature container

            sections.forEach(function (section) {
                section.style.display = 'none';
                section.style.opacity = 0;
            });

            navLinks.forEach(function (link) {
                link.classList.remove('active');
            });

            var selectedSection = document.getElementById(sectionId);
            selectedSection.style.display = 'block';
            setTimeout(function () {
                selectedSection.style.opacity = 1;
            }, 200);

            var activeLink = document.querySelector('nav ul li a[href="#' + sectionId + '"]');
            if (activeLink) {
                activeLink.classList.add('active');
            }

            // Show temperature container only when Layers section is active
            if (sectionId === 'layers') {
                temperatureContainer.style.display = 'flex';
            } else {
                temperatureContainer.style.display = 'none';
            }
        }

        let descriptionCounter = 0;

        function showDescription(layer) {
            var description = document.getElementById(layer + '-description');
            var allDescriptions = document.querySelectorAll('.description-box');
            allDescriptions.forEach(function (desc) {
                desc.style.display = 'none';
                desc.style.opacity = 0;
            });

            description.style.display = 'block';
            setTimeout(() => { description.style.opacity = 1; }, 50);

            const padding = 15;
            const descriptionHeight = description.offsetHeight;
            const maxHeight = window.innerHeight - 100;
            let offset = descriptionCounter * (descriptionHeight + padding);

            if (offset + descriptionHeight > maxHeight) {
                offset = maxHeight - descriptionHeight;
            }

            description.style.top = offset + 'px';
            descriptionCounter++;
        }
    </script>
